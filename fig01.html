def plot_figure_01(data):

    values = list(data.keys())

    figure = {
        'data'  : data[list(data.keys())[0]],
        'layout': {},
        'frames': []
    }

    figure['layout']['xaxis'] = {'title': 'Age of Pet upon Adoption (Year)', 'type':'log'}
    figure['layout']['yaxis'] = {'title': 'Adoption Rate'}
    figure['layout']['hovermode'] = 'closest'
    figure['layout']['sliders'] = \
    {
        'args': \
        [
            'transition', {
                'duration': 400,
                'easing': 'cubic-in-out'
                }
        ],
        'initialValue': values[0],
        'plotlycommand': 'animate',
        'values': values,
        'visible': True
    }
    figure['layout']['updatemenus'] = \
    [
        {
        'buttons':
            [
                {
                'args': [None, {'frame': {'duration': 500, 'redraw': False},
                                'fromcurrent': True, 'transition': {'duration': 1200, 'easing': 'quadratic-in-out'}}],
                'label': 'Play',
                'method': 'animate'
                },
                {
                'args': [[None], {'frame': {'duration': 0, 'redraw': False}, 'mode': 'immediate',
                                  'transition': {'duration': 0}}],
                'label': 'Pause',
                'method': 'animate'
                }
            ],
            'direction': 'left',
            'pad': {'r': 10, 't': 87},
            'showactive': False,
            'type': 'buttons',
            'x': 0.1,
            'xanchor': 'right',
            'y': 0,
            'yanchor': 'top'
        }
    ]

    sliders_dict = \
    {
        'active': 0,
        'yanchor': 'top',
        'xanchor': 'left',
        'currentvalue':
        {
            'font': {'size': 20},
            'prefix': 'Year:',
            'visible': True,
            'xanchor': 'right'
        },
        'transition': {'duration': 300, 'easing': 'cubic-in-out'},
        'pad': {'b': 10, 't': 50},
        'len': 0.9,
        'x': 0.1,
        'y': 0,
        'steps': []
    }

    for value in values:

        frame = {'data': data[value], 'name': str(value)}

        figure['frames'].append(frame)
        slider_step = \
        {'args': \
                [
                    [value],
                    {'frame': {'duration': 300, 'redraw': False},
                    'mode': 'immediate',
                    'transition': {'duration': 300}}
                ],
         'label': value,
         'method': 'animate'}
        sliders_dict['steps'].append(slider_step)


    figure['layout']['sliders'] = [sliders_dict]

    plot(figure, filename='figure_01.html')
