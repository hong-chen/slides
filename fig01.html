def dtime_to_jday(dtime):

    jday = (dtime - dtdt(1, 1, 1)).total_seconds()/86400.0 + 1.0

    return jday


def jday_to_dtime(jday):

    dtime = dtdt(1, 1, 1) + dt.timedelta(seconds=(jday-1)*86400.0)

    return dtime


class austin_pet_outcome:


    def __init__(self, fname):

        self.fname = fname

        vnames = ['animal_id', 'name', 'datetime', 'monthyear', 'date_of_birth', \
                  'outcome_type', 'outcome_subtype', 'animal_type', 'sex_upon_outcome', \
                  'age_upon_outcome', 'breed', 'color']

        dtype = {vname:'str' for vname in vnames}

        self.data_pd = pd.read_csv(fname, na_values=[''], skiprows=1, header=None, names=vnames, dtype=dtype)

        self.Nrow    = len(self.data_pd.index)
        self.Ncol    = len(vnames)

        self.data    = {}
        self.cal_jday()
        self.cal_age()


    def cal_jday(self):

        jday  = np.zeros(self.Nrow, dtype=np.float64)
        for i in range(self.Nrow):
            try:
                dtime0  = dtdt.strptime(self.data_pd['datetime'].iloc[i], '%m/%d/%Y %I:%M:%S %p')
                jday[i] = dtime_to_jday(dtime0)
            except:
                print('Warning [austin_pet_outcome.cal_jday]: Encounter invalid datetime string \'%s\'.' % (self.data_pd['datetime'][i]))
                jday[i] = np.nan
        self.data['jday'] = jday


    def cal_age(self):

        age  = np.zeros(self.Nrow, dtype=np.float64)
        for i in range(self.Nrow):
            try:
                dtime0  = dtdt.strptime(self.data_pd['date_of_birth'].iloc[i], '%m/%d/%Y')
                dtime1  = dtdt.strptime(self.data_pd['datetime'].iloc[i], '%m/%d/%Y %I:%M:%S %p')
                age[i]  = (dtime1-dtime0).days + 1.0
            except:
                print('Warning [austin_pet_outcome.cal_age]: Encounter invalid datetime string \'%s\'.' % (self.data_pd['date_of_birth'][i]))
                age[i] = np.nan
        self.data['age'] = age
